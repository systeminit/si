Service Creation

          
          Service.create({
            args: { 
              name: "Bugbear Graphql",
              sourceControl: { 
                repo: "systeminit/si",
                path: "components/si-graphql",
              },
              ports: [
                { serviceName: "http", 
                  name: "Bugbear GraphQL Port" },
              ]
            }
          });

Service Start

          
          Service
            .find({ name: "Bugbear Graphql" })
            .start({ server: { memoryGIB: 8000 } });

SSH Key Rotation/Rekey

          
          SshKey
            .find({ keyType: "RSA" })
            .replace({ keyType: "ED25519" });

Entity Cloning

          
          Service
            .find({ name: "Bugbear Graphql", 
                    state: "running",
                    server: { memoryGIB: 8000 } })
            .first()
            .clone(15);
